# Alphasound Configuration
# Edit this file on the SD card before inserting it.
# Changes take effect on next boot.

# Mode: RUN = car mode (WiFi AP), DEV = development mode (joins existing WiFi)
ALPHASOUND_MODE=RUN

# WiFi network name (SSID) your phone connects to
ALPHASOUND_SSID=Alphasound

# WiFi WPA2 passphrase (8-63 chars). CHANGE THIS — anyone within range
# of the AP who knows the default can connect and reach sshd.
ALPHASOUND_PASSPHRASE=alphasound

# AirPlay device name shown on your phone
ALPHASOUND_NAME=Alphasound

# WiFi channel (1-11)
ALPHASOUND_CHANNEL=11

# Country code for WiFi regulatory domain (e.g. NZ, US, GB, IE, AU)
ALPHASOUND_COUNTRY=NZ

# ALSA output device — "hw:0" for HAT DAC, "hw:1" for USB DAC
ALPHASOUND_OUTPUT_DEVICE=hw:0

# Max volume in dB (reduce if DAC overloads car AUX input, 0.0 = full)
ALPHASOUND_VOLUME_MAX_DB=-3.00

# Bluetooth A2DP receiver for non-Apple devices (yes/no)
ALPHASOUND_BLUETOOTH=yes

# SSH root password. Leave empty to disable password auth (key-only).
# CHANGE THIS — the default is well-known and the device exposes sshd
# on its WiFi AP.
ALPHASOUND_ROOT_PASSWORD=alphasound

# Development mode WiFi credentials (only used when ALPHASOUND_MODE=DEV)
ALPHASOUND_DEV_WIFI_SSID=
ALPHASOUND_DEV_WIFI_PASSWORD=

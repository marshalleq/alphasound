# Alphasound Configuration
# Edit this file on the SD card before inserting it.
# Changes take effect on next boot.

# Mode: RUN = standalone AP, DEV = client mode (joins existing WiFi).
# Can also be changed from the web UI on http://10.0.10.1/. If client
# mode fails to get an IP within 30s the device falls back to AP mode
# for that boot, so a wrong WiFi password doesn't brick it.
ALPHASOUND_MODE=RUN

# GitHub owner/repo for "Check for updates" in the web UI (client mode)
ALPHASOUND_UPDATE_REPO=marshalleq/alphasound

# WiFi network name (SSID) your phone connects to
ALPHASOUND_SSID=Alphasound

# WiFi WPA2 passphrase (8-63 chars). CHANGE THIS — anyone within range
# of the AP who knows the default can connect and reach sshd.
ALPHASOUND_PASSPHRASE=alphasound

# AirPlay device name shown on your phone
ALPHASOUND_NAME=Alphasound

# WiFi channel (1-11)
ALPHASOUND_CHANNEL=11

# Country code for WiFi regulatory domain (e.g. NZ, US, GB, IE, AU)
ALPHASOUND_COUNTRY=NZ

# ALSA output device. Also settable from the web UI Features panel.
#   auto                      Prefer DAC / USB; fall back to bcm2835 HDMI
#   hdmi                      Force HDMI (ignore any DAC that's present)
#   hw:0                      Pi onboard 3.5mm or HAT DAC at card 0
#   hw:1                      USB DAC (typical when also using onboard)
#   hw:CARD=USB               USB DAC by name (resilient to renumbering)
#   hw:CARD=sndrpihifiberry   HiFiBerry HAT DAC by name
# Run `aplay -l` over SSH to see what the kernel detected.
ALPHASOUND_OUTPUT_DEVICE=auto

# Max volume in dB (reduce if DAC overloads car AUX input, 0.0 = full)
ALPHASOUND_VOLUME_MAX_DB=-3.00

# === Optional features — disable to speed up boot ===
# Each one off saves a few seconds on boot. Safe to toggle from the web
# UI (under "Features"). Default is everything on except the display.

# Bluetooth A2DP receiver for non-Apple devices (yes/no)
ALPHASOUND_BLUETOOTH=yes

# SSH server — needed for command-line debugging, not needed for normal
# AirPlay/BT use. Off saves ~1-2s and reduces attack surface.
ALPHASOUND_SSHD=yes

# Display: shows current track + album art + progress bar when something's
# AirPlaying. "none" disables the display service entirely.
# Supported values:
#   none                     no display
#   hdmi                     render to whatever's plugged into HDMI (takes
#                            over the console for as long as it's running;
#                            switch back to "none" via the web UI to get
#                            the console back on next boot)
#   pirate-audio             Pimoroni Pirate Audio (any variant) — 240x240 ST7789
#   adafruit-1.3-tft         Adafruit 1.3" TFT (#2090) — 240x240 ST7789
#   adafruit-mini-pitft      Adafruit Mini PiTFT 1.14" (#4393) — 240x135 ST7789
ALPHASOUND_DISPLAY=none

# Roon Bridge endpoint — turns the device into a Roon-compatible audio
# zone. Requires client mode (needs to reach your Roon Core) and a
# one-time install via the web UI's "Install Roon Bridge" button (which
# downloads the binary from Roon's CDN). Roon Bridge and AirPlay both
# claim the audio device exclusively, so only one will actually play
# audio at a time — the device that most recently grabbed it wins.
ALPHASOUND_ROON=no

# SSH root password. Leave empty to disable password auth (key-only).
# CHANGE THIS — the default is well-known and the device exposes sshd
# on its WiFi AP.
ALPHASOUND_ROOT_PASSWORD=alphasound

# Development mode WiFi credentials (only used when ALPHASOUND_MODE=DEV)
ALPHASOUND_DEV_WIFI_SSID=
ALPHASOUND_DEV_WIFI_PASSWORD=
